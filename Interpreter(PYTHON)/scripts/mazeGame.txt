# Definindo variáveis
INT player_x 0
INT player_y 0

INT Aplayer_x 0
INT Aplayer_y 0

INT conf 0





FUNC NEWPOS
    Aplayer_x = player_x
    Aplayer_y = player_y

    # Configurando cor do jogador
    SET player_x player_y blue

    IF Aplayer_x notE player_x OR Aplayer_y notE player_y
        SET Aplayer_x Aplayer_y white
        PRINT Aplayer_x Aplayer_y
    ENDIF

ENDFUNC





FUNC draw
    INT dx 0
    INT dy 0
    INT dmax 10

    WHILE dx < dmax
        dy = 0
        WHILE dy < dmax
            GETMATRIXVALUE maze dy dx v
            IF v equals 1
                SET dx dy red
            ENDIF
            dy = dy + 1
        ENDWHILE
        dx = dx + 1
    ENDWHILE
ENDFUNC

# Iniciando posição do jogador
player_x = 0

Aplayer_y = player_y

CALL draw

# Funções de estado dos botões
FUNC_BTNSTATE LEFT
    IF player_x > 0
        GETMATRIXVALUE maze player_y player_x-1 cell_value
        IF cell_value equals 0
            player_x = player_x - 1
            PRINT "Movendo para a esquerda"
        ENDIF
    ENDIF
    CALL NEWPOS
ENDBTNFUNC

FUNC_BTNSTATE RIGHT
    IF player_x < 9
        GETMATRIXVALUE maze player_y player_x+1 cell_value
        IF cell_value equals 0
            player_x = player_x + 1
            PRINT "Movendo para a direita"
        ENDIF
    ENDIF
    CALL NEWPOS
ENDBTNFUNC

FUNC_BTNSTATE UP
    IF player_y > 0
        GETMATRIXVALUE maze player_y-1 player_x cell_value
        IF cell_value equals 0
            player_y = player_y - 1
            PRINT "Movendo para cima"
        ENDIF
    ENDIF
    CALL NEWPOS
ENDBTNFUNC

FUNC_BTNSTATE DOWN
    IF player_y < 9
        GETMATRIXVALUE maze player_y+1 player_x cell_value
        IF cell_value equals 0
            player_y = player_y + 1
            PRINT "Movendo para baixo"
        ENDIF
    ENDIF
    CALL NEWPOS
ENDBTNFUNC
